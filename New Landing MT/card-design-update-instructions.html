// Remove inline styles that conflict with the new design card.removeAttribute('style'); // Update labels to remove inline styles const labels = card.querySelectorAll('label'); labels.forEach(label => { label.removeAttribute('style'); }); // Update selects to remove inline styles const selects = card.querySelectorAll('select'); selects.forEach(select => { select.removeAttribute('style'); }); // Update price paragraph const pricePara = card.querySelector('p[style*="font-size: 1.3rem"]'); whatsIncluded.removeAttribute('style'); } // Update button const button = card.querySelector('.btn_prd_tab'); if (button) { button.classList.add('btn-get-tickets'); button.removeAttribute('style'); } // Update h5 const h5 = card.querySelector('h5');
